# spot mehmed failsafe. credit to foffy123
country_event = {
	id = spot.1
	title = spot.1.t
	desc = spot.1.d
	picture = CIVIL_WAR_eventPicture

	fire_only_once = yes

	trigger = {
		tag = TUR
		NOT = { has_ruler = "Mehmed II Fatih" }
		NOT = { is_year = 1475 }
	}

	 immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					is_state = yes
					is_capital = no
					is_in_capital_area = yes
				}
				save_event_target_as = spot_tur_pretenders
			}
		}
		hidden_effect = {
			random_owned_province = {
				limit = {
					is_state = yes
					is_capital = no
				}
				save_event_target_as = spot_tur_nobles
			}
		}
		hidden_effect = {
			random_owned_province = {
				limit = {
					is_state = yes
					is_capital = no
				}
				save_event_target_as = spot_tur_nobles2
			}
		}
	 }

	mean_time_to_happen = {
		months = 3
	}

	option = {
		name = spot.1.a
		add_stability = -1
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = -20
		}
		define_ruler = {
			adm = 5
			dip = 5
			mil = 5
			fixed = yes
			dynasty = ROOT
			culture = ROOT
			religion = ROOT
			age = 25
			claim = 20
		}
		event_target:spot_tur_nobles = {
			spawn_rebels = {
				type = noble_rebels
				size = 2
			}
		}
		event_target:spot_tur_nobles2 = {
			spawn_rebels = {
				type = noble_rebels
				size = 2
			}
		}
	}

	option = {
		name = spot.1.b
		add_legitimacy = 20
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = 10
		}
		event_target:spot_tur_pretenders = {
			spawn_rebels = {
				type = pretender_rebels
				size = 2
			}
		}
	}
}
